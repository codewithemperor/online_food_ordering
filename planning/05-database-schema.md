# Database Schema Planning

## 1. PostgreSQL Database Design

### 1.1 Schema Overview

The database uses PostgreSQL with Prisma ORM. All tables use CUID for primary keys (auto-generated by Prisma). Timestamps are automatically managed.

## 2. Complete Prisma Schema

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ==================== ENUMS ====================

enum Role {
  CUSTOMER
  ADMIN
}

enum UserStatus {
  ACTIVE
  BLOCKED
}

enum OrderStatus {
  PENDING
  CONFIRMED
  PREPARING
  ON_THE_WAY
  DELIVERED
  CANCELLED
}

enum PaymentMethod {
  PAYSTACK
  COD
}

enum PaymentStatus {
  PENDING
  PAID
  FAILED
  REFUNDED
}

// ==================== MODELS ====================

model User {
  id            String     @id @default(cuid())
  name          String
  email         String     @unique
  password      String
  phone         String?
  address       String?
  city          String?
  state         String?
  avatar        String?
  role          Role       @default(CUSTOMER)
  status        UserStatus @default(ACTIVE)
  createdAt     DateTime   @default(now())
  updatedAt     DateTime   @updatedAt

  orders        Order[]

  @@index([email])
  @@index([role])
  @@index([status])
  @@map("users")
}

model Category {
  id            String       @id @default(cuid())
  name          String       @unique
  description   String?
  image         String?
  createdAt     DateTime     @default(now())
  updatedAt     DateTime     @updatedAt

  restaurants   Restaurant[]
  foods         Food[]

  @@index([name])
  @@map("categories")
}

model Restaurant {
  id            String     @id @default(cuid())
  name          String
  description   String?
  email         String?
  phone         String?
  address       String?
  city          String?
  state         String?
  openTime      String?
  closeTime     String?
  openDays      String?
  image         String?
  isActive      Boolean    @default(true)
  categoryId    String
  category      Category   @relation(fields: [categoryId], references: [id], onDelete: Cascade)
  createdAt     DateTime   @default(now())
  updatedAt     DateTime   @updatedAt

  foods         Food[]
  orders        Order[]

  @@index([categoryId])
  @@index([city])
  @@index([isActive])
  @@map("restaurants")
}

model Food {
  id            String     @id @default(cuid())
  name          String
  description   String?
  price         Float
  image         String?
  isAvailable   Boolean    @default(true)
  categoryId    String
  category      Category   @relation(fields: [categoryId], references: [id], onDelete: Cascade)
  restaurantId  String
  restaurant    Restaurant @relation(fields: [restaurantId], references: [id], onDelete: Cascade)
  createdAt     DateTime   @default(now())
  updatedAt     DateTime   @updatedAt

  orderItems    OrderItem[]

  @@index([categoryId])
  @@index([restaurantId])
  @@index([isAvailable])
  @@index([price])
  @@map("foods")
}

model Order {
  id              String           @id @default(cuid())
  orderNumber     String           @unique
  userId          String
  user            User             @relation(fields: [userId], references: [id])
  restaurantId    String?
  restaurant      Restaurant?      @relation(fields: [restaurantId], references: [id])
  status          OrderStatus      @default(PENDING)
  subtotal        Float
  deliveryFee     Float            @default(0)
  totalAmount     Float
  paymentMethod   PaymentMethod    @default(PAYSTACK)
  paymentStatus   PaymentStatus    @default(PENDING)
  paystackRef     String?
  deliveryAddress String?
  deliveryCity    String?
  deliveryState   String?
  customerPhone   String?
  customerName    String?
  deliveryInstructions String?
  remark          String?
  cancelledAt     DateTime?
  deliveredAt     DateTime?
  createdAt       DateTime         @default(now())
  updatedAt       DateTime         @updatedAt

  items           OrderItem[]
  statusHistory   OrderStatusHistory[]

  @@index([userId])
  @@index([restaurantId])
  @@index([status])
  @@index([paymentStatus])
  @@index([orderNumber])
  @@index([createdAt])
  @@map("orders")
}

model OrderItem {
  id            String   @id @default(cuid())
  orderId       String
  order         Order    @relation(fields: [orderId], references: [id], onDelete: Cascade)
  foodId        String
  foodName      String
  foodImage     String?
  quantity      Int      @default(1)
  unitPrice     Float
  totalPrice    Float

  @@index([orderId])
  @@index([foodId])
  @@map("order_items")
}

model OrderStatusHistory {
  id            String      @id @default(cuid())
  orderId       String
  order         Order       @relation(fields: [orderId], references: [id], onDelete: Cascade)
  status        OrderStatus
  remark        String?
  changedBy     String?
  createdAt     DateTime    @default(now())

  @@index([orderId])
  @@map("order_status_history")
}
```

## 3. Index Strategy

### 3.1 Primary Indexes (Auto-created)
- All `id` fields have primary key indexes
- All `@unique` fields have unique indexes (email, orderNumber, category name)

### 3.2 Performance Indexes

| Table | Column | Index Type | Reason |
|-------|--------|-----------|--------|
| users | email | Unique | Login lookup |
| users | role | B-tree | Admin queries |
| users | status | B-tree | Active/blocked filter |
| categories | name | Unique | Lookup by name |
| restaurants | categoryId | B-tree | Filter by category |
| restaurants | city | B-tree | Filter by city |
| restaurants | isActive | B-tree | Active restaurants only |
| foods | categoryId | B-tree | Filter by category |
| foods | restaurantId | B-tree | Filter by restaurant |
| foods | isAvailable | B-tree | Available foods only |
| foods | price | B-tree | Price range queries |
| orders | userId | B-tree | User's orders |
| orders | restaurantId | B-tree | Restaurant's orders |
| orders | status | B-tree | Filter by status |
| orders | paymentStatus | B-tree | Filter by payment |
| orders | orderNumber | Unique | Lookup by order number |
| orders | createdAt | B-tree | Sort by date, date range |
| order_items | orderId | B-tree | Order's items |
| order_items | foodId | B-tree | Food's order items |
| order_status_history | orderId | B-tree | Order's status history |

### 3.3 Composite Indexes (Future)
- (status, createdAt) on orders — for dashboard queries
- (categoryId, isAvailable) on foods — for menu filtering
- (restaurantId, isAvailable) on foods — for restaurant menu

## 4. Data Integrity Rules

### 4.1 Cascade Deletes
```
Category → Restaurant (Cascade: Delete category deletes all restaurants)
Restaurant → Food (Cascade: Delete restaurant deletes all foods)
Order → OrderItem (Cascade: Delete order deletes all items)
Order → OrderStatusHistory (Cascade: Delete order deletes all history)
```

### 4.2 Restrict Deletes
```
User → Order (Restrict: Cannot delete user with orders — recommend blocking)
Category with only 1 remaining (Business rule: Cannot delete last category)
```

### 4.3 Snapshot Pattern
Order items store food data at the time of ordering:
- `foodName` — Copied from Food.name at order time
- `foodImage` — Copied from Food.image at order time
- `unitPrice` — Copied from Food.price at order time (server-verified)

This ensures that:
- Food price changes don't affect past orders
- Food deletion doesn't affect past orders
- Food name changes don't affect past orders

## 7. Query Patterns & Performance

### 7.1 Common Queries

**Get foods with filters:**
```sql
SELECT f.*, c.name as category_name, r.name as restaurant_name
FROM foods f
JOIN categories c ON f."categoryId" = c.id
JOIN restaurants r ON f."restaurantId" = r.id
WHERE f."isAvailable" = true
  AND ($1::text IS NULL OR f."categoryId" = $1)
  AND ($2::text IS NULL OR f."restaurantId" = $2)
  AND ($3::text IS NULL OR f.name ILIKE '%' || $3 || '%')
ORDER BY f.name ASC
LIMIT 20 OFFSET 0;
```

**Get user orders with items:**
```sql
SELECT o.*, 
  json_agg(json_build_object(
    'id', oi.id,
    'foodName', oi."foodName",
    'quantity', oi.quantity,
    'unitPrice', oi."unitPrice",
    'totalPrice', oi."totalPrice"
  )) as items
FROM orders o
LEFT JOIN order_items oi ON o.id = oi."orderId"
WHERE o."userId" = $1
GROUP BY o.id
ORDER BY o."createdAt" DESC
LIMIT 20;
```

**Dashboard stats:**
```sql
-- Single efficient query for multiple stats
SELECT
  (SELECT COUNT(*) FROM restaurants) as total_restaurants,
  (SELECT COUNT(*) FROM foods) as total_foods,
  (SELECT COUNT(*) FROM users WHERE role = 'CUSTOMER') as total_customers,
  (SELECT COUNT(*) FROM orders) as total_orders,
  (SELECT COUNT(*) FROM orders WHERE status = 'PENDING') as pending_orders,
  (SELECT COUNT(*) FROM orders WHERE DATE("createdAt") = CURRENT_DATE) as today_orders,
  (SELECT COALESCE(SUM("totalAmount"), 0) FROM orders WHERE "paymentStatus" = 'PAID') as total_earnings,
  (SELECT COUNT(*) FROM orders WHERE status = 'CANCELLED') as cancelled_orders;
```

**Earnings by period:**
```sql
SELECT DATE("createdAt") as date, 
  SUM("totalAmount") as revenue, 
  COUNT(*) as orders
FROM orders 
WHERE "paymentStatus" = 'PAID'
  AND "createdAt" >= $1
GROUP BY DATE("createdAt")
ORDER BY date DESC;
```

### 7.2 N+1 Prevention
- Use Prisma `include` for related data (categories, restaurants, order items)
- Use `select` for specific fields when full data not needed
- Batch queries with `Promise.all` for dashboard stats
- Paginate all list endpoints
